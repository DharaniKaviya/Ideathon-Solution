# Clone repository
git clone https://github.com/yourusername/rural-healthcare-system.git
cd rural-healthcare-system

# Install dependencies
pip install -r requirements.txt

# Setup database
mysql -u root -p < database/schema.sql

# Configure environment
cp .env.example .env
# Edit .env with your database credentials

# Run backend
python app.py

# Open frontend
# Visit: http://localhost:5000/frontend/
# Or open frontend/index.html in browser
